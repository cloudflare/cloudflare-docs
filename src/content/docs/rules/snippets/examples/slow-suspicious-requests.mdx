```js
export default {
    async fetch(request) {
        // Define delay
        const delay_in_seconds = 5;
        // Introduce a delay
        await new Promise(resolve => setTimeout(resolve, delay_in_seconds * 1000)); // Set delay in milliseconds

        // Pass the request to the origin
        const response = await fetch(request);
        return response;
    },
};
```
